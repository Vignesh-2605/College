# Experiment 4
# Plural Generator

def plural(word):

    if word.endswith(("s", "x", "z", "ch", "sh")):
        return word + "es"

    elif word.endswith("y") and word[-2] not in "aeiou":
        return word[:-1] + "ies"

    else:
        return word + "s"


word = input("Enter a noun: ")

print("Singular :", word)
print("Plural   :", plural(word))

# Experiment 9: Rule-Based POS Tagging

import nltk
from nltk.tag import RegexpTagger

patterns = [
    (r'.*ing$', 'VBG'),
    (r'.*ed$', 'VBD'),
    (r'.*es$', 'VBZ'),
    (r'.*ould$', 'MD'),
    (r'.*\'s$', 'NN$'),
    (r'.*s$', 'NNS'),
    (r'^-?[0-9]+(.[0-9]+)?$', 'CD'),
    (r'.*', 'NN')
]

tagger = RegexpTagger(patterns)

sentence = input("Enter a sentence: ")

words = sentence.split()

print("\nTagged Output:")

for word, tag in tagger.tag(words):
    print(word, ":", tag)

# Experiment 11: Top-Down Parser

from nltk.parse import TopDownChartParser

S -> NP VP
NP -> Det N
VP -> V NP
Det -> 'the'
N -> 'cat' | 'mouse'
V -> 'chased'
""")

# Experiment 14: Subject-Verb Agreement

sentence = input("Enter a sentence: ")

words = sentence.split()

if len(words) >= 2:
    subject = words[0]
    verb = words[1]

    if subject.lower() in ["he", "she", "it"] and verb.endswith("s"):
        print("Agreement Correct")

    elif subject.lower() in ["i", "you", "we", "they"] and not verb.endswith("s"):
        print("Agreement Correct")

    else:
        print("Agreement Incorrect")
else:
    print("Invalid Sentence")

# Experiment 21: Syntax-Driven Semantic Analysis

import nltk

nltk.download('punkt')
nltk.download('averaged_perceptron_tagger')

sentence = input("Enter a sentence: ")

tokens = nltk.word_tokenize(sentence)

tags = nltk.pos_tag(tokens)

grammar = "NP: {<DT>?<JJ>*<NN.*>+}"

chunk_parser = nltk.RegexpParser(grammar)

tree = chunk_parser.parse(tags)

print(tree)

print("\nNoun Phrases:")

for subtree in tree.subtrees():
    if subtree.label() == 'NP':
        print(" ".join(word for word, tag in subtree.leaves()))

# Experiment 23: Text Coherence Evaluation

import nltk
from nltk.tokenize import sent_tokenize, word_tokenize

# Download required package (only first time)
nltk.download('punkt')

text = input("Enter a paragraph:\n")

sentences = sent_tokenize(text)

print("\nNumber of Sentences:", len(sentences))

for i, sentence in enumerate(sentences, start=1):
    words = word_tokenize(sentence)
    print(f"Sentence {i}: {len(words)} words")

# Simple coherence check
if len(sentences) > 1:
    print("\nThe text appears to be coherent.")
else:
    print("\nThe text may not be coherent.")
