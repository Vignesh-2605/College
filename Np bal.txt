print("\n--- 2. Finite State Automaton (ending with 'ab') ---")

class FSAEndsWithAB:
    def __init__(self):
        self.state = 'q0'

    def transition(self, char):
        if self.state == 'q0':
            self.state = 'q1' if char == 'a' else 'q0'
        elif self.state == 'q1':
            self.state = 'q2' if char == 'b' else ('q1' if char == 'a' else 'q0')
        elif self.state == 'q2':
            self.state = 'q1' if char == 'a' else ('q2' if char == 'b' else 'q0')

    def recognizes(self, input_string):
        self.state = 'q0'
        for char in input_string:
            self.transition(char)
        return self.state == 'q2'

fsa = FSAEndsWithAB()

print(f"'cab' ends with 'ab': {fsa.recognizes('cab')}")
print(f"'abc' ends with 'ab': {fsa.recognizes('abc')}")
print(f"'abab' ends with 'ab': {fsa.recognizes('abab')}")
print(f"'a' ends with 'ab': {fsa.recognizes('a')}")
print(f"'b' ends with 'ab': {fsa.recognizes('b')}")
print(f"'' ends with 'ab': {fsa.recognizes('')}")
print(f"'aaab' ends with 'ab': {fsa.recognizes('aaab')}")
print(f"'ca' ends with 'ab': {fsa.recognizes('ca')}")

print("\n--- 4. Finite-State Machine for Pluralization ---")

class PluralFSM:
    def pluralize(self, noun):
        irregulars = {
            'man': 'men',
            'woman': 'women',
            'child': 'children',
            'tooth': 'teeth',
            'foot': 'feet',
            'mouse': 'mice',
            'goose': 'geese'
        }

        if noun in irregulars:
            return irregulars[noun]

        if noun.endswith('y') and len(noun) > 1 and noun[-2] not in 'aeiou':
            return noun[:-1] + 'ies'

        if noun.endswith(('ch', 'sh', 'x', 's')):
            return noun + 'es'

        if noun.endswith('o'):
            return noun + 'es'

        return noun + 's'

fsm = PluralFSM()

nouns = [
    'cat', 'dog', 'box', 'bush',
    'church', 'sky', 'potato',
    'man', 'apple', 'day', 'boy'
]

for noun in nouns:
    print(f"Singular: {noun}, Plural: {fsm.pluralize(noun)}")

import random
from collections import defaultdict

print("\n--- 6. N-gram Model (Bigram) for Text Generation ---")

def build_bigram_model(text):
    words = text.lower().split()
    model = defaultdict(list)

    for i in range(len(words) - 1):
        model[words[i]].append(words[i + 1])

    return model

def generate_text(model, start_word, num_words=10):
    current_word = start_word.lower()
    generated_text = [current_word]

    for _ in range(num_words - 1):
        if current_word in model and model[current_word]:
            current_word = random.choice(model[current_word])
            generated_text.append(current_word)
        else:
            break

    return ' '.join(generated_text)

corpus = "I love natural language processing. Natural language processing is fascinating. I really love this."

bigram_model = build_bigram_model(corpus)

print("Corpus:", corpus)
print("Generated text (starting with 'i'):", generate_text(bigram_model, 'i', 8))
print("Generated text (starting with 'processing'):", generate_text(bigram_model, 'processing', 7))

import random
from collections import defaultdict

print("\n--- 8. Simple Stochastic Part-of-Speech Tagging ---")

class SimpleStochasticPOSTagger:

    def __init__(self):
        self.word_tag_counts = defaultdict(lambda: defaultdict(int))
        self.tag_counts = defaultdict(int)

    def train(self, corpus_sentences):
        for sentence in corpus_sentences:
            for word, tag in sentence:
                self.word_tag_counts[word.lower()][tag] += 1
                self.tag_counts[tag] += 1

    def tag(self, words):
        tagged_sentence = []

        for word in words:
            word_lower = word.lower()
            possible_tags = self.word_tag_counts[word_lower]

            if not possible_tags:
                tagged_sentence.append((word, 'NN'))
                continue

            best_tag = max(possible_tags, key=possible_tags.get)
            tagged_sentence.append((word, best_tag))

        return tagged_sentence

training_corpus = [
    [('The', 'DT'), ('dog', 'NN'), ('barks', 'VBZ'), ('.', '.')],
    [('A', 'DT'), ('cat', 'NN'), ('meows', 'VBZ'), ('.', '.')],
    [('The', 'DT'), ('cats', 'NNS'), ('sleep', 'VBP'), ('.', '.')],
    [('The', 'DT'), ('bird', 'NN'), ('sings', 'VBZ'), ('.', '.')]
]

st_tagger = SimpleStochasticPOSTagger()
st_tagger.train(training_corpus)

text_to_tag = "The cats bark .".split()

import re

print("\n--- 9. Rule-Based Part-of-Speech Tagging (Regular Expressions) ---")

class RegexPOSTagger:

    def __init__(self):
        self.rules = [
            (r'.*ing$', 'VBG'),
            (r'.*ed$', 'VBD'),
            (r'.*es$', 'VBZ'),
            (r'.*s$', 'NNS'),
            (r'.*ly$', 'RB'),
            (r'.*ful$', 'JJ'),
            (r'^[A-Z].*$', 'NNP'),
            (r'the|a|an', 'DT'),
            (r'.*', 'NN')
        ]

    def tag(self, words):
        tagged_sentence = []

        for word in words:
            found_tag = False

            for pattern, tag_label in self.rules:
                if re.match(pattern, word, re.IGNORECASE):
                    tagged_sentence.append((word, tag_label))
                    found_tag = True
                    break

            if not found_tag:
                tagged_sentence.append((word, 'UNKNOWN'))

        return tagged_sentence

print("\n--- 10. Transformation-Based Tagging (Simple Rule) ---")

class TransformationTagger:
    def __init__(self, initial_tagger): self.initial_tagger = initial_tagger; self.transformation_rules = []

    def add_rule(self, condition_func, transform_func): self.transformation_rules.append((condition_func, transform_func))

    def tag(self, words):
        initial_tagged = self.initial_tagger.tag(words) if self.initial_tagger else [(word, 'NN') for word in words]
        current_tagged_sentence = list(initial_tagged)

        for i in range(len(current_tagged_sentence)):
            word, current_tag = current_tagged_sentence[i]
            for condition_func, transform_func in self.transformation_rules:
                if condition_func(i, word, current_tag, current_tagged_sentence):
                    current_tagged_sentence[i] = (word, transform_func(i, word, current_tag, current_tagged_sentence)); break
        return current_tagged_sentence

class BasicInitialTagger:
    def tag(self, words):
        tagged = []
        for word in words:
            if word.endswith('s'): tagged.append((word, 'NNS'))
            elif word in ['the', 'a']: tagged.append((word, 'DT'))
            elif word.endswith('ed'): tagged.append((word, 'VBD'))
            else: tagged.append((word, 'NN'))
        return tagged

def condition_nn_to_vb(idx, word, current_tag, tagged_sentence):

print("\n--- 11. Top-Down (Recursive Descent) Parser for CFG ---")

grammar = {
    'S': [['NP', 'VP']], 'NP': [['Det', 'N']], 'VP': [['V', 'NP'], ['V']],
    'Det': [['the'], ['a']], 'N': [['dog'], ['cat']], 'V': [['saw'], ['ran']]
}

def parse(symbol, tokens, pos):
    if symbol not in grammar:
        return pos + 1 if pos < len(tokens) and tokens[pos] == symbol else None
    for production in grammar[symbol]:
        p = pos
        for sym in production:
            p = parse(sym, tokens, p)
            if p is None: break
        if p is not None: return p
    return None

for sentence in ["the dog saw a cat", "a cat ran", "dog the saw"]:
    tokens = sentence.split()
    end = parse('S', tokens, 0)
    print(f"'{sentence}' -> {'VALID' if end == len(tokens) else 'INVALID'}")

print("\n--- 12. Earley Parser for CFG ---")

grammar = {
    'S': [['NP', 'VP']], 'NP': [['Det', 'N']], 'VP': [['V', 'NP']],
    'Det': [['the']], 'N': [['dog'], ['cat']], 'V': [['saw']]
}

def earley(words):
    chart = [[] for _ in range(len(words) + 1)]
    chart[0].append(('S', ['S'], 0, 0))
    for i in range(len(words) + 1):
        k = 0
        while k < len(chart[i]):
            lhs, rhs, dot, start = chart[i][k]; k += 1
            if dot < len(rhs):
                nxt = rhs[dot]
                if nxt in grammar:
                    for prod in grammar[nxt]:
                        st = (nxt, prod, 0, i)
                        if st not in chart[i]: chart[i].append(st)
                elif i < len(words) and words[i] == nxt:
                    chart[i+1].append((lhs, rhs, dot+1, start))
            else:
                for st in chart[start]:
                    if st[2] < len(st[1]) and st[1][st[2]] == lhs:
                        new = (st[0], st[1], st[2]+1, st[3])
                        if new not in chart[i]: chart[i].append(new)
    return ('S', ['S'], 1, 0) in chart[len(words)]

for s in ["the dog saw the cat", "the cat saw dog"]:
    print(f"'{s}' -> {'ACCEPTED' if earley(s.split()) else 'REJECTED'}")

print("\n--- 18. Simple FOPC (First-Order Predicate Calculus) Parser ---")

import re

def parse_fopc(expr):
    expr = expr.strip()
    m = re.match(r'(forall|exists)\s+(\w+)\s+(.*)', expr)
    if m:
        return {'quantifier': m.group(1), 'variable': m.group(2), 'body': parse_fopc(m.group(3))}
    for conn in ['->', '&', '|']:
        if conn in expr:
            l, r = expr.split(conn, 1)
            return {'connective': conn, 'left': parse_fopc(l), 'right': parse_fopc(r)}
    m = re.match(r'(\w+)\((.*)\)', expr)
    if m:
        return {'predicate': m.group(1), 'args': [a.strip() for a in m.group(2).split(',')]}
    return {'atom': expr}

for s in ["forall x Man(x) -> Mortal(x)", "exists y Loves(john, y)"]:
    print(f"\nInput : {s}\nParsed: {parse_fopc(s)}")

from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
import nltk

print("\n--- 23. Text Coherence Evaluation ---")

def coherence(text):
    sents = nltk.sent_tokenize(text)
    vecs = TfidfVectorizer().fit_transform(sents)
    sims = [cosine_similarity(vecs[i], vecs[i+1])[0][0] for i in range(len(sents)-1)]
    return sum(sims)/len(sims), sims

for label,text in [
    ("Coherent","The cat is a pet. The cat likes to drink milk. Milk is good for the cat."),
    ("Incoherent","The cat is a pet. Quantum physics is hard. I bought a car yesterday.")
]:
    score,sims = coherence(text)
    print(f"{label}: avg coherence = {score:.3f} (adjacent sims: {[round(s,3) for s in sims]})")

print("\n--- 24. Dialog Act Recognition ---")

import re

def dialog_act(u):
    t = u.lower().strip()
    if re.search(r'\b(hi|hello|hey|good morning)\b', t): return "GREETING"
    if t.endswith('?') or re.match(r'(what|why|how|when|where|who|is|are|do|can|could)\b', t): return "QUESTION"
    if re.search(r'\b(thanks|thank you)\b', t): return "THANKING"
    if re.search(r'\b(bye|goodbye|see you)\b', t): return "FAREWELL"
    if re.search(r'\b(yes|sure|ok|okay|yeah)\b', t): return "AGREEMENT"
    if re.search(r'\b(no|nope|not)\b', t): return "REJECTION"
    return "STATEMENT"

dialog = [
    "Hello there!",
    "How are you doing?",
    "I am fine.",
    "Can you help me?",
    "Yes sure.",
    "Thanks a lot.",
    "Goodbye!"
]

for u in dialog:
    print(f"{u:<20} -> {dialog_act(u)}")
